def overcounting(x) :
    freq = {}
    for i in x :
        if i not in freq.keys():
            freq[i] = 1
        else :
            freq[i] += 1
    
    oc = 1
    
    for f in freq.values():
        oc *= MF(f)
    
    return(oc)
