#Clear axes.
def clearAx(ax):
    """
    Clear axes.

    Parameters:
    ax - Current ax.
    
    Returns:
    ax - Updated ax.
    """    
    
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    ax.spines["bottom"].set_visible(False)
    ax.spines["left"].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])
    return(ax)




def createSemiCircle() :
    """
    Get coordinates for semicircle.
    
    Parameters:
    None
    
    Returns:
    (X, Y) - Vector of two lists.
    """
    
    X = [x/250 for x in range(-250, 251, 1)]
    
    diam = []
    semi = []
    
    for x in X:
        diam.append(0)
        semi.append(mS(1-x**2))
        
    X += X[::-1][1:]
    Y = diam + semi[::-1][1:]
    
    return(X, Y)
    


    
def getAuxillaries(ax, o, k, x1, y1, x2, y2, x3, y3, x4, y4, x5, y5) :
    """
    Adds auxillary dotted lines for solving the problem.

    Parameters:
        ax - Current ax.
        o - Fiddler or EC tag.
        k - Constant of ±1.
        x1, y1 - 10:05 wave endpoint.
        x2, y2 - Intersection point of perpendicular from 10:05 wave to first contact.
        x3, y3 - Intersection point of perpendicular from 10:05 wave to last contact.
        x4, y4 - 10:05 wave endpoint.
        x5, y5 - Point of last contact with wave.
    
    Returns:
    ax - Updated ax.
    """
    
    def auxillaryHelper(ax, x, y) :
        ax.plot(x, y,
                c = c[5],
                lw = 2,
                zorder = 4,
                ls = ':')
        #return(ax)
    
    if o == "Fiddler" :
        auxillaryHelper(ax, [0, x1], [0, y1])
        auxillaryHelper(ax, [2*x3-x5, x5], [2*y3-y5, y5])
        
    elif o == "EC1" :
        auxillaryHelper(ax, [-1, x2], [0, y2])
        auxillaryHelper(ax, [0, x1], [0, y1])
        auxillaryHelper(ax, [0, x5], [0, y5])
        
    elif o == "EC2" :
        auxillaryHelper(ax, [2*x3-x5, x5], [2*y3-y5, y5])
        auxillaryHelper(ax, [0, x4], [0, y4])
        auxillaryHelper(ax, [0, x1], [0, y1])
        
    return(ax)




def getFirstLastContact(ax, k, ex, ey, x5, y5) :
    """
    Adds the first and last contact of the wave.

    Parameters:
        ax - Current ax.
        k - Constant of ±1.
        ex, ey - Extend lines based off angle.
        x5, y5 - Point of last contact with wave.
    
    Returns:
    ax - Updated ax.
    """     
    
    #First.
    ax.plot([k + ex, k - ex], [ey, -ey], c = c[2], lw = 4, zorder = 3)
    #Last
    ax.plot([x5 + ex, x5 - ex], [y5 + ey, y5 - ey], c = c[2], lw = 4, zorder = 3)
    return(ax)




def get1005(ax, ex, ey, x1, y1, x4, y4) :
    """
    Get the wave through the island at 10:05.

    Parameters:
        ax - Current ax.
        ex, ey - Extend lines based off angle.
        x1, y1 - 10:05 wave endpoint.
        x4, y4 - 10:05 wave endpoint.
    
    Returns:
    ax - Updated ax.
    """     
    
    ax.plot([x1+ex, x1], [y1+ey, y1], c = c[4], lw = 5, ls = ':', zorder = 2)
    ax.plot([x1, x4], [y1, y4], c = c[4], lw = 5, zorder = 2)
    ax.plot([x4, x4-ex], [y4, y4-ey], c = c[4], lw = 5, ls = ':', zorder = 2)
    return(ax)

def semicircleIsland(fig, o, i) :

    #Create two subplots.
    gs = fig.add_gridspec(nrows = 1,
                          ncols = 2,
                          width_ratios = [11, 3],
                          wspace = 0)
    
    ax1 = fig.add_subplot(gs[0, 0])
    ax1.set_xlim(-1.1, 1.1)
    ax1.set_ylim(-0.5, 1.1)
    ax2 = fig.add_subplot(gs[0, 1])
    ax2.set_xlim(1.1, 1.7)
    ax2.set_ylim(-0.5, 1.1)
    
    #Remove axes and ticks.
    ax1 = clearAx(ax1)
    ax2 = clearAx(ax2)
    
    #Degrees to Radians
    t = i*pi/180
    
    #Symmetry!
    if t <= pi/2 :
        k = -1

    elif t <= pi :
        k = 1

    #All the constants!    
    a = (1 + k*cos(t)) / 2
    b = a / cos(t)    
    u = t - k*invcos(a)     
    x1 = cos(u)
    y1 = sin(u)
    x2 = k + (1 - a) * cos(t)
    y2 = (1 - a) * sin(t)
    x3 = a*cos(t)
    y3 = a*sin(t)
    x4 = b
    y4 = 0
    x5 = cos(t)
    y5 = sin(t)
    ex = 3*cos(pi/2+t)
    ey = 3*sin(pi/2+t)
    
    if (pi/2 - invsin(1/3) < t) and (t < pi/2 + invsin(1/3)) :
        x4 = cos(2*t-u)
        y4 = sin(2*t-u)

    #Ocean backdrop
    ax1.add_patch(mpR(xy = (-1.1, -1.1),
                      width = 2.2,
                      height = 2.2,
                      ec = None,
                      fc = c[3],
                      alpha = 0.4,
                      zorder = 1))
  
    #Semicircle Island
    X, Y = createSemiCircle()
    semicircle = np.column_stack((X, Y))
    ax1.add_patch(mpP(xy = semicircle, ec = c[0], fc = c[1], lw = 4, zorder = 1))
    ax1.plot(X, Y, c = c[0], lw = 4, zorder = 3)
    
    #First and last xontact waves.
    ax1 = getFirstLastContact(ax1, k, ex, ey, x5, y5)

    #Get wave @ 10:05.
    ax1 = get1005(ax1, ex, ey, x1, y1, x4, y4)
    
    #Images for solving.
    if o != 'EC' :
        
        #Auxillaries.
        ax1 = getAuxillaries(ax1, o, k, x1, y1, x2, y2, x3, y3, x4, y4, x5, y5)
        
        if o == 'Fiddler' :

            #Vertices.
            ax1.scatter([x1, x3, x4, x5, -1, 0, 1],
                        [y1, y3, y4, y5, 0, 0, 0],
                        c = 'k',
                        lw = 3,
                        zorder = 5)
            #Annotations.
            text = [('1', (-0.5, -0.05), 'k', 20, 0),
                    ('1', (-0.4, 0.2), 'k', 20, 0),
                    ('1', (0.5, -0.05), 'k', 20, 0),
                    ('α', (0.05, 0.05), 'k', 20, 0),
                    ('cos α', (-0.17, -0.25), 'k', 15, t*180/pi),
                    ('cos α', (-0.06, 0.09), 'k', 15, t*180/pi),
                    ('2cos α', (0.1, 0.54), 'k', 15, t*180/pi),
                    ('sin α', (-0.43, 0.48), 'k', 15, t*180/pi-90),
                    ('sin α', (0.5, 0.16), 'k', 15, t*180/pi-90),]

        elif o == "EC1" :

            ax1.scatter([x1, x2, x3, x4, x5, -1, 0, 1],
                        [y1, y2, y3, y4, y5, 0, 0, 0],
                        c = 'k',
                        lw = 3,
                        zorder = 5)

            text = [('α', (0.1, 0.05), 'k', 20, 0),
                    ('A', (1, -0.10), 'k', 20, 0),
                    ('B', (b, -0.10), 'k', 20, 0),
                    ('C', (0, -0.10), 'k', 20, 0),
                    ('D', (-1, -0.10), 'k', 20, 0),
                    ('E', (x1, y1 + 0.05), 'k', 20, 0),
                    ('F', (x5, y5 + 0.05), 'k', 20, 0),
                    ('G', (x3, y3 + 0.05), 'k', 20, 0),
                    ('H', (x2, y2 - 0.10), 'k', 20, 0)]

        elif o == "EC2" :

            ax1.scatter([x1, 2*x3-x5, x3, x4, x5, -1, 0, 1],
                        [y1, 2*y3-y5, y3, y4, y5, 0, 0, 0],
                        c = 'k',
                        lw = 3,
                        zorder = 5)

            text = [('A', (x1, y1 + 0.05), 'k', 20, 0),
                    ('B', (1, -0.15), 'k', 20, 0),

                    ("C'", (0, -0.2), 'k', 20, 0),

                    ('E', (x4, y4 + 0.05), 'k', 20, 0),
                    ('F', (x5, y5 + 0.05), 'k', 20, 0),
                    ('G', (x3, y3 + 0.05), 'k', 20, 0)]

        for tt in text :
            ax1.annotate(text = tt[0],
                         xy = tt[1],
                         c = tt[2],
                         size = tt[3],
                         rotation = tt[4],
                         zorder = 5)
            
        fig.savefig("2026.06.12" + o + ".png", bbox_inches = "tight")   

    #Animation.    
    else :
        
        d = mS((x4-x1)**2 + (y4-y1)**2)
        
        text = [('Angle :', (1.55, 0.8), c[2], 40, 'center',  0),
                ("{:.1f}".format(i) + '°', (1.55, 0.55), c[2], 35, 'center', 0 ),
                ('Halfway', (1.55, 0.2), c[4], 40, 'center', 0),
                ('Stretch :', (1.57, 0), c[4], 40, 'center', 0),
                ("{:.3f}".format(d), (1.55, -0.25), c[4], 35, 'center', 0 )]

        for tt in text :
            ax2.annotate(text = tt[0],
                         xy = tt[1],
                         c = tt[2],
                         size = tt[3],
                         ha = tt[4],
                         rotation = tt[5],
                         zorder = 5)

def catchTheWave() :
    
    fig = plt.figure(figsize = (10.5, 6))

    #Run animation.
