def clearAx(ax):
    '''
    Clears all axes and tick marks.
    '''
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    ax.spines["bottom"].set_visible(False)
    ax.spines["left"].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])
    return(ax)



def arcCreator(X, Y, R, T1, T2) :
    '''
    Creates an arc.
    
    Input: (X,Y) - Center of circle
           R - radius
           T1 - Start of arc, angle.
           T2 - End of arc, angle. Direction is counterclockwise.
           
    Output: Two lists, each containing 201 X and Y corresponding to T1 to T2 along circle circumference.
    '''
    
    arcX = []
    arcY = []
    for i in range(201):
        
        step = (T2-T1) / 200
        arcX.append(X + R*cos(T1 + i*step))
        arcY.append(Y + R*sin(T1 + i*step))
        
    return(arcX, arcY)

def fiddlerPseudoSquare(op):
    '''
    '''
    fig = plt.figure(figsize = (9, 9))
    ax = fig.add_subplot(xlim = (-0.3, 1.2),
                         ylim = (-0.4, 1.1))
    clearAx(ax)
    #ax.patch.set_facecolor('w')
    
    r = 0.18387
    t = 0.84468
    
    #Left small-circle arc.
    s_x, s_y =  arcCreator(0, 0, r, t, 2*pi)
    ax.plot(s_x,
            s_y,
            lw = 1.5,
            c = 'k')
    
    #Right, large-circle arc.    
    l_x, l_y =  arcCreator(0, 0, 1+r, 0, t)
    ax.plot(l_x,
            l_y,
            lw = 1.5,
            c = 'k')
    
    #Bottom segment.
    ax.plot([s_x[-1], l_x[0]],
            [s_y[-1], l_y[0]],
            lw = 1.5,
            c = 'k')    
    #Top segment.
    ax.plot([s_x[0], l_x[-1]],
            [s_y[0], l_y[-1]],
            lw = 1.5,
            c = 'k')
    
    #Diagram.
    if op == 1:
    
        #Radii.
        ax.plot([s_x[-1], 0, s_x[0]],
                [s_y[-1], 0, s_y[0]],
                lw = 2,
                ls = '--',
                c = '#AAAAAAFF') 
        
        
        
        text = [("r", (s_x[-1]/2, -0.02), 'center', 'top'),
                ("ɸ", (0.07, 0.035), 'center', 'center'),
                ("1", (r*cos(pi +t/2), r*sin(pi +t/2)), 'right', 'top'),
                ("1", ((s_x[-1]+l_x[0])/2, -0.02), 'center', 'top'),
                ("1", ((1+r)*cos(t/2), (1+r)*sin(t/2)), 'left', 'bottom')]
    
        for t in text :
            plt.annotate(text = t[0],
                         xy = t[1],
                         c = 'k',
                         size = 16,
                         ha = t[2],
                         va = t[3],
                         zorder = 4)
    
    #Area & Color.
    else :

        #Append center and first point.
        s_x = s_x + [0] + [s_x[0]]
        s_y = s_y + [0] + [s_y[0]]
        #Transpose to coordinate pairs.
        s_xy = np.array([s_x, s_y]).T
        #Color in.
        ax.add_patch(mpP(s_xy,
                         ec = v[1],
                         fc = v[1]))
        
        #Append center and first point.
        l_x = l_x + [0] + [l_x[0]]
        l_y = l_y + [0] + [l_y[0]]
        #Transpose to coordinate pairs.
        l_xy = np.array([l_x, l_y]).T
        #Color in.
        ax.add_patch(mpP(l_xy,
                         ec = v[2],
                         fc = v[2]))
        
        
        
    fig.savefig("2024.11.01fiddler" + str(op) + ".png", bbox_inches = 'tight')
