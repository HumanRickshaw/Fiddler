#Polygon pieces of cake.
def polygonPiece(points, edgecolor, facecolor):
    
    return(mpP(points,
               edgecolor = edgecolor,
               facecolor = facecolor,
               lw = 3))

def cakeCuts(option) :
    
    fig = plt.figure(figsize = (10, 5))
    ax = fig.add_subplot(xlim = (-2.02, 2.02),
                         ylim = (-1.01, 1.01))

    #Remove axes and ticks.
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    ax.spines["bottom"].set_visible(False)
    ax.spines["left"].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])
    
    #Rectangle corners.
    c1 = (-2, -1)
    c2 = (-2, 1)
    c3 = (2, 1)
    c4 = (2, -1)
    
    #Frostings.
    f = ['#2A788EFF', '#2A788E66',
         '#440154FF', '#44015466',
         '#7AD151FF', '#7AD15166']
    
    if option == 1:
        
        center = (0, 1/3)
        #Ends of 3 cuts.
        es = [c1, (0, 1), c4]

        #3 Polygon pieces of cake.
        piece1 = polygonPiece([c1, center, c4, c1], f[0], f[1])
        piece2 = polygonPiece([c1, c2, es[1], center, c1], f[2], f[3])
        piece3 = polygonPiece([center, es[1], c3, c4, center], f[4], f[5])
       
    if option == 2:
        
        center = (1/3, 0)
        #Ends of 3 cuts.
        es = [(-2, 0), (1, 1), (1, -1)]
        
        #3 Polygon pieces of cake.
        piece1 = polygonPiece([c1, es[0], center, es[2], c1], f[0], f[1])
        piece2 = polygonPiece([es[0], c2, es[1], center, es[0]], f[2], f[3])
        piece3 = polygonPiece([es[2], center, es[1], c3, c4, es[2]], f[4], f[5])


    if option == 3:
        
        center = (5/39, 8/39)
        #Ends of 3 cuts.
        es = [(-2, -.5), (.5, 1), (1.5, -1)]

        #3 Polygon pieces of cake.
        piece1 = polygonPiece([c1, es[0], center, es[2], c1], f[0], f[1])
        piece2 = polygonPiece([es[0], c1, c2, es[1], center, es[0]], f[2], f[3])
        piece3 = polygonPiece([es[2], center, es[1], c3, c4, es[2]], f[4], f[5])
        
    ax.add_patch(piece1)
    ax.add_patch(piece2)
    ax.add_patch(piece3)
    
    #Slices are black.
    for p in es :
        plt.plot([p[0], center[0]],
                 [p[1], center[1]],
                 c = 'k',
                 lw = 4,
                 zorder = 2)       

    
    if option == 3:
        
        #Origin.
        plt.scatter(0, 0, c = 'k', lw = 7)
        #New Center.    
        plt.scatter(center[0], center[1], c = 'grey', lw = 7, zorder = 3)
        
        #Horizontal & Vertical
        plt.plot([-1.99, 1.99],
                 [center[1], center[1]],
                 'grey',
                 lw = 2,
                 ls = '--',
                 zorder = 1)
        
        plt.plot([center[0], center[0]],
                 [-0.99, 0.99],
                 'grey',
                 lw = 2,
                 ls = '--',
                 zorder = 1)
        
        text = [#Regions
            ("A1", (-0.6, -0.65), f[0], 36, 0),
            ("A2", (0.45, -0.65), f[0], 36, 0),
            ("B1", (center[0], 0.65), f[2], 36, 0),
            ("B2", (-1.65, -0.1), f[2], 36, 0),
            ("C1", (1.3, 0.5), f[4], 36, 0),
            ("C2", (1.3, -0.5), f[4], 36, 0),

  
            #Around the perimeter.
            ("z", (-1.92, -0.75), 'k', 28, 90),
            ("5+y-z", (-1.92, (-0.5+center[1])/2), 'k', 14, 90),
            ("5-y", (-1.92, (1+center[1])/2), 'k', 14, 90),
            
            ("10+x", ((-2+center[0])/2, 0.92), 'k', 14, 0),
            ("z", ((0.5+center[0])/2-0.06, 0.92), 'k', 28, 0),
            ("-x", ((0.5+center[0])/2+0.03, 0.92), 'k', 14, 0),
            ("10-z", (1.25, 0.92), 'k', 14, 0),
            
            ("5-y", (1.92, (1+center[1])/2), 'k', 14, 270),
            ("5+y", (1.92, (-1+center[1])/2), 'k', 14, 270),
            
            ("z", (1.75, -.92), 'k', 28, 0),
            ("10-x-z", ((1.5+center[0])/2, -.92), 'k', 14, 0),
            ("10+x", ((-2+center[0])/2, -.92), 'k', 14, 0),
        
            #Origin & Center.
            ("(0, 0)", (-0.3, -0.2), 'k', 28, 0),
            ("(x, y)", (center[0]+0.4, center[1]+0.15), 'gray', 28, 0),]
            
        #Pow.
        for t in text :
            plt.annotate(text = t[0],
                         xy = t[1],
                         c = t[2],
                         size = t[3],
                         ha = "center",
                         va = "center",
                         rotation = t[4],
                         zorder = 2)
            
    fig.savefig("2023.03.01" + str(option) + ".png", bbox_inches = 'tight')
