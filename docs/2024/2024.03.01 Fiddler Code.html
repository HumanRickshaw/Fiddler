def polygonPiece(points, edgecolor, facecolor):
    
    return(mpP(points,
               edgecolor = edgecolor,
               facecolor = facecolor,
               lw = 3))

def clearAx(ax):
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    ax.spines["bottom"].set_visible(False)
    ax.spines["left"].set_visible(False)
    ax.set_xticks([])
    ax.set_yticks([])
    return(ax)

def X(z):
    x = (-15*z**2 + 150*z) / (3*z**2 - 30*z + 300)
    return(x)


#y in terms of x
def Y(x):
    y = math.sqrt((-3*x + 5) * (x + 5))/3
    return(y)

zs = [z/20 for z in range(101)]

def init():
    """Initialize seven artists."""
    ax.add_patch(p1)
    ax.add_patch(p2)
    ax.add_patch(p3)
    
    ax.add_patch(s1)
    ax.add_patch(s2)
    ax.add_patch(s3)
    
    ax.add_patch(center_sh)
    
    z_text.set_text('')
    xy_text.set_text('')
    
    return p1, p2, p3, s1, s2, s3, center_sh, z_text, xy_text

def animate(i):

    z = zs[i]
    x = xs[i]
    y = ys[i]
    center = centers[i]
    end = ends[i]
    
    """Update nine artists."""
    if z <= 2 :
        p1.set_xy([c[0], end[0], center, end[2], c[0]])
        p2.set_xy([end[0], c[1], end[1], center, end[0]])
        p3.set_xy([center, end[1], c[2], c[3], end[2], center])
    else :
        p1.set_xy([c[0], c[1], end[0], center, end[2], c[0]])
        p2.set_xy([center, end[0], c[2], end[1], center])
        p3.set_xy([end[2], center, end[1], c[3], end[2]])

    
    s1.set_xy([end[0], center])
    s2.set_xy([end[1], center])
    s3.set_xy([end[2], center])

    center_sh.set_xy(centers[0:(i+1)])

    return p1, p2, p3, s1, s2, s3, center_sh, z_text, xy_text
