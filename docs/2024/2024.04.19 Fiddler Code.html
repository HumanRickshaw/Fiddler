def getDeck():
    deck = []
    for m in range(13):
        deck.append(str(m))
    deck = deck * 4
    return(deck)

def collCardG(deck) :
    
    p = len(deck)/2
    
    while len(deck) > 0 :
        if deck.pop() == deck.pop() :
            deal = p - len(deck)/2
            return("Loss", deal)
    
    return("Win", p+0.1)

def monteCarlo(n, t) :
    record = []
    for i in range(n):
        deck = getDeck()
        if t == 'ec':
            deck *= 2
        RS(deck)
        record.append(collCardG(deck)[1])
        
    freqDist = {}
    for r in record :
        if r in freqDist.keys():
            freqDist[r] += 1
        else :
            freqDist[r] = 1
                
    freqDist = dict(sorted(freqDist.items()))
    df = pd.DataFrame(data = {"Round": freqDist.keys(),
                              "Frequency": freqDist.values(),
                              "Outcome": ["Loss"]*(len(freqDist)-1)+["Win"]})
    
    df.loc[len(df)-1, 'Round'] -= 0.1
    return(df)
