#Initialize Common Neighbors Dictionary.
#(R0, C0) : [(R0, C1), (R1, C0),...].
#Player R0, C0 is neighbors with cells in the list.
def initializeTerritoryAndCommonEdge(N) :
    
    #Get the neighbors, if they exist.  Count each neighbor once.
    def getNeighbors(r, c) :
        
        neighbors = set([])
        
        #Left.
        z=0 if c == 0 else neighbors.add((r, c-1))
        
        #Top
        z=0 if r == 0 else neighbors.add((r-1, c))
        
        #Right.
        z=0 if c == N-1 else neighbors.add((r, c+1))
        
        #Bottom
        z=0 if r == N-1 else neighbors.add((r+1, c))
        
        return(neighbors)
    
    #Territory held.
    T = {}
    #Neighbors.
    CE = {} 
    
    for r in range(N):
        for c in range(N):
            #Every N^2 player is named by its initial cell territory.
            T[(r, c)] = {(r,c)}
            CE[(r, c)] = getNeighbors(r, c)
    
    return(T, CE)





#Append the P_k's territory to that of P_j.
#Remove P_k.
def updateTerritory(j, k, T) :

    T[j] = T[j] | T[k]
    del T[k]

    return(T)





#1) For each non-winner neighbor of the loser P_k, replace P_k with P_j.
#2) Append the P_k's neighbors to that of P_j.  Remove P_k and P_j from that list.
#3) Remove P_k.
def updateCommonEdge(j, k, CE) :
    
    #Neighbors of k.
    temp = CE[k]
    
    #1)
    #Non-winner neighbors of 'Pk'.
    for n in temp :
        
        if n == j :
            pass 
        
        else :
            #Neighbors of that neighbor.
            ce_n = CE[n]
            #Remove 'Pk', add 'Pj'.
            ce_n.remove(k)
            ce_n.add(j)
            CE[n] = ce_n
      
    #2)
    CE[j] = CE[j] | CE[k]
    CE[j].remove(k)
    CE[j].remove(j)
    
    #3)
    del CE[k]
    
    return(CE)




#Play!
def theFloor(N) :
    
    #Initialize
    T, CE = initializeTerritoryAndCommonEdge(N)

    #Play
    Round = N*N
    
    #Eliminate one player each round.  THERE CAN BE ONLY ONE!
    while Round > 1 :
        
        #Get two players from those players in common_edges.  
        temp = RS(list(CE.keys()), 1)[0]
        #Randomly select winner/loser.
        wl = RS([temp, RS(list(CE[temp]), 1)[0]], 2)
        j = wl[0]
        k = wl[1]

        #Update and continue.
        T = updateTerritory(j, k, T)
        CE = updateCommonEdge(j, k, CE)
        Round -= 1

    return(j)





def monteCarlo(M, N) :

    result = {}
    for m in range(M) :
        n = theFloor(N)
        if n in result.keys():
            result[n] += 1
        else :
            result[n] = 1
    
    for k in result.keys():
        result[k] = round(result[k] / M, 5)
    

    return(result)

def graphPrep(results) :
    
    def symmetryCleanUp(points, r, div) :
        div_up = 0
        for op in points :
            div_up += r[op]
        div_up = div_up/div
        for op in points :
            r[op] = div_up
        return(r)
    
    #Final output.
    df = pd.DataFrame(columns = ['N', 'X', 'Y', 'P'])
    
    for i, r in enumerate(results):
        N = i+2
        for x in range(MC(N/2)) :
            for y in range(x+1) :
                div_up = 0
                
                #Diagonals, only 4.
                #Center is only 1 but repeated 4 times if N is odd.
                if x == y :
                    r = symmetryCleanUp([(x, x), (x, N-1-x), (N-1-x, N-1-x), (N-1-x, x)], r, 4)
                                
                #Odd vertical and horizontal, only 4.
                elif x == N/2-1/2 :
                    r = symmetryCleanUp([(x, y), (y, x), (x, N-1-y), (N-1-y, x)], r, 4)
                
                #All other points, come in octets.
                else :
                    r = symmetryCleanUp([(x, y), (y, x), (y, N-1-x), (x, N-1-y), (N-1-x, N-1-y),
                                         (N-1-y, N-1-x), (N-1-y, x), (N-1-x, y)], r, 8)
    
        for k in r.keys() :
            df.loc[len(df)] = [i+2, k[0], k[1], r[k]]
    
    df.sort_values(by = ['N', 'X', 'Y'], inplace = True)
    return(df)
